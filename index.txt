(function() {
  // Ajustes
  const DAYS_AFTER = 2;

  // Hora del recordatorio (elige una que NO sea 23:59).
  // Yo pongo 12:00 para minimizar ‚Äúest√° durmiendo / liada‚Äù.
  const REMINDER_HOUR = 12;
  const REMINDER_MIN  = 0;

  // Duraci√≥n del evento (en minutos)
  const DURATION_MIN = 10;

  // Textos del evento
  const TITLE = "Dejar rese√±a de Atr√°palo (show)";
  const DETAILS = "Busca el email de Atr√°palo (puede caer en Promociones/Spam) y deja la rese√±a. Gracias ‚ù§Ô∏è";

  // Calcula fecha local: ahora + 2 d√≠as, a las 12:00
  const now = new Date();
  const start = new Date(now);
  start.setDate(start.getDate() + DAYS_AFTER);
  start.setHours(REMINDER_HOUR, REMINDER_MIN, 0, 0);

  const end = new Date(start);
  end.setMinutes(end.getMinutes() + DURATION_MIN);

  // Mostrar al usuario (hora local del dispositivo)
  const fmt = new Intl.DateTimeFormat('es-ES', { dateStyle: 'full', timeStyle: 'short' });
  document.getElementById('when').textContent =
    "üìÖ Recordatorio: " + fmt.format(start) + " (hora de tu m√≥vil)";

  // Helpers formato Google Calendar: YYYYMMDDTHHMMSS (local) + Z en UTC es m√°s robusto
  function toUTCCompact(d) {
    const pad = n => String(n).padStart(2,'0');
    const y = d.getUTCFullYear();
    const m = pad(d.getUTCMonth()+1);
    const da = pad(d.getUTCDate());
    const h = pad(d.getUTCHours());
    const mi = pad(d.getUTCMinutes());
    const s = pad(d.getUTCSeconds());
    return `${y}${m}${da}T${h}${mi}${s}Z`;
  }

  const dates = `${toUTCCompact(start)}/${toUTCCompact(end)}`;

  // Link Google Calendar
  const gcalUrl = new URL("https://calendar.google.com/calendar/render");
  gcalUrl.searchParams.set("action", "TEMPLATE");
  gcalUrl.searchParams.set("text", TITLE);
  gcalUrl.searchParams.set("details", DETAILS);
  gcalUrl.searchParams.set("dates", dates);

  document.getElementById('gcal').href = gcalUrl.toString();

  // Generar ICS (para Apple Calendar)
  function toICSDateUTC(d) {
    // YYYYMMDDTHHMMSSZ
    return toUTCCompact(d);
  }

  const icsContent =
`BEGIN:VCALENDAR
VERSION:2.0
PRODID:-//Recordatorio Atrapalo//ES
CALSCALE:GREGORIAN
METHOD:PUBLISH
BEGIN:VEVENT
UID:${crypto.randomUUID()}
DTSTAMP:${toICSDateUTC(new Date())}
DTSTART:${toICSDateUTC(start)}
DTEND:${toICSDateUTC(end)}
SUMMARY:${TITLE}
DESCRIPTION:${DETAILS.replace(/\n/g, "\\n")}
END:VEVENT
END:VCALENDAR`;

  document.getElementById('ics').addEventListener('click', () => {
    const blob = new Blob([icsContent], { type: "text/calendar;charset=utf-8" });
    const url = URL.createObjectURL(blob);
    const a = document.createElement('a');
    a.href = url;
    a.download = "recordatorio-atrapalo.ics";
    document.body.appendChild(a);
    a.click();
    a.remove();
    URL.revokeObjectURL(url);
  });
})();
